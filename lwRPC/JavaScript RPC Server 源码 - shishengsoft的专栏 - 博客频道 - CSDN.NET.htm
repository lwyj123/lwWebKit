<?php



/**

 * JavaScript RPC Server

 * @class    JsRPC

 * @author   xie sheng tao <xie.s.t@163.com>

 * @date     2008-03-28  2008-04-02  2008-7-3

 */

class JsRPC

{

    /**

     * 注册的Server端函数名列表

     * @var  array   $m_funcList

     */

    private $m_funcList = array();



    /**

     * Server端函数执行结果

     * @var  any   $m_responseData       PHP支持的几种基本数据类型

     */

    private $m_responseData = null;

    

    /**

     * 构造函数

     */

    public function __construct()

    {

        ob_end_clean();

    }

    /**

     * 注册Server端函数

     * @param    string  $funcName   函数名

     * @return   null

     */

    public function register($funcName)

    {

        $this->m_funcList[$funcName] = $funcName;

    }



    /**

     * 开始响应客户端请求，执行相应的函数

     * @param    null

     * @return   null

     */

    public function start()

    {

        $exeResult = '';

        if($this->getFunctionInfo($funcName, $parameter) 

            && $this->isRegister($funcName))

        {

            try

            {

               /**

                * 执行函数，根据页面输出的错误信息，抛出相应的异常。

                * 如果页面有任何输出，也将抛出异常。

                */

                ob_start();

                eval('$this->m_responseData='.$funcName.'('.$parameter.');');

                $exeResult = ob_get_contents();

                ob_end_clean();



                if($exeResult == '')//没有异常，输出执行结果。

                {

                    $jsSerial = $this->php2JsSerialize($this->m_responseData);

                    ob_start();

                    echo "SERVER_OK/n";

                    echo '('.$jsSerial.')';

                    ob_flush();

                    exit();

                }

                else//执行错误，输出页面信息

                {

                    echo "SERVER_ERROR/n".strip_tags($exeResult);

                    exit();

                }

            }

            catch(Exception $ex)

            {

                echo "SERVER_ERROR/n";

                echo($ex->getMessage());

                exit();

            }

        }else{

            echo "SERVER_ERROR/n";

            echo('Invoke funciton ['.$funcName.'] is inexistent.');

            exit();

        }

    }



    /**

     * 取得PHP的数组和对象的JSON脚本字符串

     * @param    array or object or base type   $varArrayObj

     * @return   string

     */

    private function php2JsSerialize($phpVar)

    {

        $type = gettype($phpVar);

        //特殊类型

        if($type == 'NULL') return ('null');

        //四种标量类型

        if($type == 'boolean') return ($phpVar ? 'true' : 'false');

        if($type == 'integer') return ($phpVar);

        if($type == 'double') return ($phpVar);

        if($type == 'string') return ('"' . addslashes($phpVar) . '"');

        //两种复合类型

        if($type == 'array' || $type == 'object')

        {

            $phpScript = '{';

            foreach($phpVar as $key=>$val)

            {

                $phpScript .= ''.$key.':'.$this->php2JsSerialize($val).',';

            }

            $phpScript .= '}';

            $phpScript = str_replace(',}', '}', $phpScript);

            return $phpScript;

        }

        //其它类型

        return ('null');

    }



    /**

     * 获取Ajax提交的函数名和函数参数

     * @param    string  &$funcName

     * @param    string  &$paramater

     * @reurn    boolean

     */

    private function getFunctionInfo(&$funcName, &$parameter)

    {

        if(isset(REQUEST['funcName'])){

            $funcName = stripslashes(REQUEST['funcName']);

        }

        if(isset(REQUEST['parameter'])){

            $parameter = stripslashes(REQUEST['parameter']);

        }

        if($funcName){

            return true;

        }

        return false;

    }



    /**

     * 判断客户端请求执行的服务端函数是否已经注册到JsRPC之中。

     * @param    string  $funcName

     * @return   boolean

     */

    private function isRegister($funcName)

    {

        return (isset($this->m_funcList[$funcName]));

    }



} // end class



?>
